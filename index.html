// 客户端信息（多 IP API 轮询 + 重试按钮 + 错误提示） const clientInfoEl = document.getElementById('clientInfo'); const refreshBtn = document.getElementById('refreshClientInfo'); const ua = navigator.userAgent; const screenInfo = `${screen.width}x${screen.height}`; const ipApis = [ { url: 'https://api.ipify.org?format=json', extract: d => JSON.parse(d).ip }, { url: 'https://api.ip.sb/ip', extract: d => d.trim() }, { url: 'https://api64.ipify.org?format=json', extract: d => JSON.parse(d).ip }, ]; async function fetchWithTimeout(url, timeout = 3000) { const controller = new AbortController(); const id = setTimeout(() => controller.abort(), timeout); const res = await fetch(url, { signal: controller.signal }); clearTimeout(id); return res; } async function loadClientInfo() { clientInfoEl.innerHTML = `UA: ${ua}
屏幕: ${screenInfo}
IP: 获取中...`; let lastError = ''; for (const api of ipApis) { try { const res = await fetchWithTimeout(api.url); const text = await res.text(); const ip = api.extract(text); if (ip && /^[\d.]+$/.test(ip)) { clientInfoEl.innerHTML = `IP: ${ip}
UA: ${ua}
屏幕: ${screenInfo}`; return; } } catch (e) { lastError = e.message; continue; } } clientInfoEl.innerHTML = `IP: ❌ 获取失败（${lastError || '所有服务无响应'}）
UA: ${ua}
屏幕: ${screenInfo}`; } loadClientInfo(); refreshBtn.addEventListener('click', loadClientInfo);
