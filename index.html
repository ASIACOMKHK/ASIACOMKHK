1. 边缘计算的优势
Cloudflare Workers 让代码运行在 300+ 边缘节点，无需维护服务器。NetDiag 利用这一特性实现了全球低延迟网络探测。

// NetDiag 核心探测示例
async function handleRequest(request) {
  const url = new URL(request.url);
  if (url.pathname === "/ping") {
    return new Response("pong", { status: 200 });
  }
  // DNS 解析示例
  const dns = await fetch("https://cloudflare-dns.com/dns-query?...");
  return new Response(await dns.text());
}
